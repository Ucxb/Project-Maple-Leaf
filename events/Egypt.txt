#Egypt: The Issue of Halaib
country_event = {
	id = egypt.1
	title = egypt.1.t
	desc = egypt.1.d
	picture = GFX_report_event_halaib
	
	fire_only_once = yes
	
	trigger = {
		tag = EGY
		date > 2000.1.10
		date < 2002.1.1
		798 = {
			is_controlled_by = SUD
		}
		NOT = { has_country_flag = egy_issue_of_halab }
	}
	
	immediate = {
		set_country_flag = egy_issue_of_halab
	}
	
	option = { #Press the Issue
		name = egypt.1.o1
		SUD = {
			country_event = {
				id = egypt.2
			}
			add_opinion_modifier = {
				target = EGY
				modifier = recent_actions_negative
			}
		}
		ai_chance = {
			factor = 90
		}	
	}
	
	option = { #Back down
		name = egypt.1.o2
		798 = {
			remove_core_of = EGY
		}
		SUD = {
			add_opinion_modifier = {
				target = EGY
				modifier = recent_actions_very_positive
			}
		}
		add_political_power = -100
		add_stability = -0.15
		set_country_flag = egy_backed_down_from_halaib
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		hidden_effect = {
			news_event = {
				id = news.44
				hours = 6
			}
		}
	}
}
