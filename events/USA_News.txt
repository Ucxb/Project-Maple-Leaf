		}
	}
	
	option = {
		name = usa_news.18.o3

			OR = {

			}

	option = {
		name = usa_news.18.o4
		trigger = {
			original_tag = SUD

				add_opinion_modifier = {

					modifier = recent_actions_negative
				}
			}
		}
	}
